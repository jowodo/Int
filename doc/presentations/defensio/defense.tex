			\begin{itemize}

	s_i = \begin{cases}
		1 &\text{if} \quad g_i \geq 10^{-5} \\
		0 &\text{if} \quad g_i < 10^{-5} \\

%        0 &\text{if} \quad -log_{10}(g_i) \geq 5 \\
	\end{cases}
					$
				\item $
	\rho = \sum_i^N \frac{s_i}{N}
					$
			\end{itemize}
	\vspace{0.5em}
%	\pause
		\item Leakage Current 
			\begin{itemize}
				\item $

	$
			\end{itemize}
	\end{itemize}
